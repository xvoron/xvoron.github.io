import numpy as np
import matplotlib.pyplot as plt
import math

def user_input_process():
    """User input processing function.
    This field is the multi-line type of commentaries.
    When you use it in functions or classes this called "Doc-string".

    If you write: >help(user_input_process)
    This doc string will show up.

    Arguments:
        Here you can write some function arguments.

        Example:
        argument_1 - 1 input argument.

    Return:
        And here add some return values commentary.

        user_input_function - user input function <sin/cos>.

        user_input_freq - user input frequency <0.1-20>.

    """

    # Single line commentary

    # Simply way to check user input.
    while True:
        user_input_function = input("Select a function <sin/cos>: ")
        if user_input_function == "sin" or user_input_function == "cos":
            break
        else:
            continue

   # Better way. Using "try-except" construction 
   # Check input frequency.
    while True:
        try:
            user_input_freq = int(input("Enter a frequency <1-20>: "))
        except ValueError:
            print("[ERROR] your input must be integer in range <1-20>.")
            continue
        if user_input_freq >= 1 and user_input_freq <= 20:
            break
        else:
            continue

    return user_input_function, user_input_freq

def function_generator(func_type, freq):
    """Generate data for input sin or cos function.

    Arguments:
        func_type - type of function <sin/cos>.
        freq - value of frequency.

    Return:
        t - time series.
        y - y-axis values
    """

    t = np.linspace(0,3,100*freq)    # Create a time series using numpy 

    if func_type == "sin":
        y = np.sin(2*np.pi*freq*t)
    else:
        y = np.cos(2*np.pi*freq*t)

    # Dictionary data type
    dictionary_of_parametrs = {"function" : func_type,
                                "frequency" : freq}

    """Print summary data in table.
    Example of using "for" with dictionaries.
        | key1 | value1 |
        | key2 | value2 |
    """
    print("| Parameter | Value |")
    for key, value in dictionary_of_parametrs.items():
        print("| " + str(key)   + " "*(9-len(key))        + \
              "| " + str(value) + " "*(6-len(str(value))) + "|")

    return t, y


def function_plot(x, y, func_type):
    """Plot user input function.

    Arguments:
        x - x-axis values.
        y - y-axis values.
    """

    fig, ax = plt.subplots()            # Create figure and axis

    ax.plot(x, y, label=r"y={0}(x)".format(func_type))    # Plot
    ax.legend()                         # Add legend
    ax.set_xlabel("t")                  # Set x,y labels
    ax.set_ylabel("y")
    ax.set_title("Function")            # Add Title

    plt.show()                          # Show figure

if __name__ == "__main__":

    function_type, frequency = user_input_process()
    x, y = function_generator(function_type, frequency)
    function_plot(x, y, function_type)
